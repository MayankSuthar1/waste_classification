    # Waste Classification Dataset

# Run from project root: python setup_dataset.py

# 1. Metal
# 2. Carton
# 3. Glass
# 4. Organic Waste
# 5. Other Plastics
# 6. Paper and Cardboard
# 7. Plastic
# 8. Textiles
# 9. Wood

# Total: ~5,078 images
# Split: 90% training (4,571 images), 10% validation (507 images)
